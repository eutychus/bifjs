Use with clappr

You can add video thumbnails from BIF files using the clappr player and the clappr thumbnails plugin. Here's a quick code sample.
if(bifurl) { console.log(bifurl); thisbif = new BIFReader(bifurl); thisbif.load(function(bif_reader, idx) { console.log('bif loaded'); console.log([thisbif.count, thisbif.multiplier]); var thumbnailsPlugin = player.getPlugin("scrub-thumbnails"); for(var i = 0; i < thisbif.count; i++) { thumbnailsPlugin.addThumbnail({url: thisbif.idx[i][1], time:(i*thisbif.multiplier/1000)}); } }, function(progress) { console.log(progress); }); }
